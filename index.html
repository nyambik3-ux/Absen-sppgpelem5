// ========== FUNGSI TANGGAL YANG LEBIH KONSISTEN ========== function getTodayKey() { const d = new Date(); const year = d.getFullYear(); const month = (d.getMonth() + 1).toString().padStart(2, '0'); const day = d.getDate().toString().padStart(2, '0'); return `${year}-${month}-${day}`; } const STORAGE_KEY = `absen_${user.id}_${getTodayKey()}`; // (Opsional) Ubah sessionStorage ke localStorage jika ingin status tetap tersimpan saat browser ditutup let sudahMasuk = localStorage.getItem(STORAGE_KEY) === 'masuk'; let sudahPulang = localStorage.getItem(STORAGE_KEY) === 'pulang'; // ========== FUNGSI ABSEN DENGAN GEOLOCATION ========== async function doAbsen(action) { btnMasuk.disabled = true; btnPulang.disabled = true; // Ambil lokasi pengguna secara otomatis (jika didukung dan diizinkan) let lat = 0; let lng = 0; try { const position = await new Promise((resolve, reject) => { navigator.geolocation.getCurrentPosition(resolve, reject); }); lat = position.coords.latitude; lng = position.coords.longitude; } catch (err) { console.warn("Gagal mendapatkan lokasi atau izin ditolak. Menggunakan koordinat default."); } try { const response = await fetch(SCRIPT_URL, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ action: action, // "DATANG" atau "PULANG" id: user.id, nama: user.nama || user.id, lat: lat, lng: lng }) }); const data = await response.json(); if (data.result === "success") { if (action === "DATANG") { localStorage.setItem(STORAGE_KEY, 'masuk'); sudahMasuk = true; showToast("✅ Absen Masuk berhasil!"); } else if (action === "PULANG") { localStorage.setItem(STORAGE_KEY, 'pulang'); sudahPulang = true; showToast("✅ Absen Pulang berhasil!"); } updateButtonState(); if (sudahPulang) { setTimeout(() => window.location.reload(), 1500); } } else { showToast("❌ Gagal: " + (data.message || "Coba lagi"), true); updateButtonState(); } } catch (err) { console.error(err); showToast("❌ Gagal koneksi ke server", true); updateButtonState(); } } // ========== EVENT LISTENER LOGOUT ========== document.getElementById('logoutBtn').addEventListener('click', () => { sessionStorage.removeItem('user_pelem_5'); localStorage.removeItem(STORAGE_KEY); window.location.href = "login.html"; });
